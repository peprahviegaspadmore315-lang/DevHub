def slugify(name):
    s=name.lower().strip()
    s=s.replace('/','-')
    s=s.replace('__','-').replace('...','').replace('..','')
    for ch in ['?','!','\"','+','#',':','@','%','&','(',')',',']:
        s=s.replace(ch,'')
    s=s.replace(' ','-')
    while '--' in s: s=s.replace('--','-')
    s=s.strip('-')
    return s
